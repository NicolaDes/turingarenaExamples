// nC = number of clauses
// nL = number of literals
// cnf[][] = sat formula
// cnf3[][] = 3-sat formula (to fill)
// certificate[][] = 3-sat certificate

procedure reduce(sizeL,literals[],sizeC,clause[]) callbacks {
    procedure addVariable(l);
}

main {

    read sizeL;
    for i to sizeL {
        read literals[i];
    }
    read sizeC;
    for i to sizeC {
        read clause[i];
    }


    call reduce(sizeL,literals, sizeC, clause) callbacks {
        procedure addVariable(l) {
            write l;
        }
    }
}
