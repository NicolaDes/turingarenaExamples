// nC = number of clauses
// nL = number of literals
// cnf[][] = sat formula
// cnf3[][] = 3-sat formula (to fill)
// certificate[][] = 3-sat certificate

function reduce(size,clause[]);
function getLiteral(i);

main {

    read literals;
    for i to literals {
        read clause[i];
    }

    call size = reduce(literals,clause);

    write size;

    for i to size {
        call lit = getLiteral(i);
        write lit;
    }

}
