function invita(int N, int M, int[] conoscenzeA, int[] conoscenzeB) -> int;

main {
    var int N;
    var int M;
    var int[] conoscenzeA;
    var int[] conoscenzeB;

    var int ans;

    read N;
    read M;

    alloc conoscenzeA, conoscenzeB : M;
    
    for(e : M)
    {
        read conoscenzeA[e];
        read conoscenzeB[e];
    }

    call invita(N,M,conoscenzeA,conoscenzeB) -> ans;

    write ans;
}
